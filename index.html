Python Heart Drawing with Turtle

Here is the Python code to draw a heart for Valentine's Day:

import turtle

pen = turtle.Turtle()

def curve():
    for i in range(200):
        pen.right(1)
        pen.forward(1)

def heart():
    pen.fillcolor('red')
    pen.begin_fill()
    pen.left(140)
    pen.forward(113)
    curve()
    pen.left(120)
    curve()
    pen.forward(112)
    pen.end_fill()

def txt():
    pen.up()
    pen.setpos(0, 95)
    pen.down()
    pen.color('lightgreen')
    pen.write("З Днем святого Валентина", align="center", font=("Verdana", 12, "bold"))

heart()
txt()
pen.ht()
turtle.done()
